using System.Linq;

public class SmartServerSelector
{
    public enum ServerRegion { USEast, USCentral, USWest }
    
    public class PlayerLatency
    {
        public string PlayerID;
        public int PingToEast;
        public int PingToCentral;
        public int PingToWest;
    }
    
    public ServerRegion SelectOptimalServer(List<PlayerLatency> players)
    {
        // Calculate worst-case ping for each server option
        int worstPingEast = players.Max(p => p.PingToEast);
        int worstPingCentral = players.Max(p => p.PingToCentral);
        int worstPingWest = players.Max(p => p.PingToWest);
        
        // Choose server with lowest worst-case ping
        // This ensures no player suffers extremely
        
        if (worstPingCentral <= worstPingEast && worstPingCentral <= worstPingWest)
        {
            return ServerRegion.USCentral; // Most balanced
        }
        else if (worstPingEast <= worstPingWest)
        {
            return ServerRegion.USEast;
        }
        else
        {
            return ServerRegion.USWest;
        }
    }
}
